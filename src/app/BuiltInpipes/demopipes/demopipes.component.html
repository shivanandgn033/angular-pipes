demopipes works!


1. DatePipe :

{{ today | date }}


{{ today | date:'fullDate' }}


{{ today | date:'shortTime' }}



UpperCasePipe and LowerCasePipe :

{{ 'Hello World' | uppercase }}


{{ 'Hello World' | lowercase }}



SlicePipe :

{{ 'Hello World' | slice:0:5 }}


{{ 'Hello World' | slice:6:11 }}



JsonPipe :

{{ { name: 'John', age: 30 } | json }}



CurrencyPipe :

{{ price | currency }}


{{ price | currency:'USD':true }}


{{ price | currency:'USD':false }}



PercentPipe :

{{ 0.5 | percent }}


{{ 0.5 | percent:'2.1-2' }}



DecimalPipe :

{{ 0.5 | number }}


{{ 0.5 | number:'2.1-2' }}



AsyncPipe :

{{ promise | async }}



KeyValuePipe :
  • {{ item.key }} : {{ item.value }}

TitleCasePipe :

{{ 'hello world' | titlecase }}



I18nPluralPipe :

{{ count }} {{ count | i18nPlural: { '=0': 'No messages.', '=1': 'One message.', 'other': '# messages.' } }}
